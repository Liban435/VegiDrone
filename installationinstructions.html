Create a database and run the following commands:

  • DROP TABLE IF EXISTS `admin`;
    CREATE TABLE IF NOT EXISTS `admin` (
    `uname` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    `pass_hash` char(60) COLLATE utf8_unicode_ci NOT NULL,
    PRIMARY KEY (`uname`)
    ) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

  • DROP TABLE IF EXISTS `admin_tokens`;
    CREATE TABLE IF NOT EXISTS `admin_tokens` (
    `uname` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    `admin_token` char(32) COLLATE utf8_unicode_ci NOT NULL,
    PRIMARY KEY (`uname`)
    ) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

  • DROP TABLE IF EXISTS `banned_emails`;
    CREATE TABLE IF NOT EXISTS `banned_emails` (
    `id` int(11) NOT NULL AUTO_INCREMENT,
    `email` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    PRIMARY KEY (`id`)
    ) ENGINE=MyISAM AUTO_INCREMENT=4 DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

  • DROP TABLE IF EXISTS `user`;
    CREATE TABLE IF NOT EXISTS `user` (
    `email` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    `pass_hash` char(60) COLLATE utf8_unicode_ci NOT NULL,
    `fname` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    `lname` varchar(64) COLLATE utf8_unicode_ci NOT NULL,
    PRIMARY KEY (`email`)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

  • DROP TABLE IF EXISTS `user_tokens`;
    CREATE TABLE IF NOT EXISTS `user_tokens` (
    `email` varchar(60) COLLATE utf8_unicode_ci NOT NULL,
    `user_token` char(32) COLLATE utf8_unicode_ci NOT NULL,
    PRIMARY KEY (`email`)
    ) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
